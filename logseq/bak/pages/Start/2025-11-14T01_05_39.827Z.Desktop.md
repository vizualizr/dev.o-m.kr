- ## Updates
	- #+BEGIN_QUERY
	  { :query [:find (pull ?p [:block/name :page/date-modified])
	           :where
	           ; 1. 페이지를 찾고 이름을 가져옵니다.
	           [?p :block/name ?name]
	           
	           ; 2. date-modified 속성을 가진 페이지를 필터링하고 그 값을 ?modified-date에 바인딩합니다.
	           ; 이 속성 값은 반드시 "YYYY-MM-DD" 형태의 일반 문자열이어야 합니다.
	           [?p :page/date-modified ?modified-date]
	           
	           ; 3. 제외할 페이지 목록을 필터링합니다.
	           (not [(contains? #{"Start" "contents" "favorites" "recent"} ?name)])]
	           
	   :result-transform (fn [result]
	                       (->> result
	                            ; 4. :page/date-modified 값을 기준으로 내림차순 정렬 (가장 최신 날짜가 위로)
	                            (sort-by :page/date-modified >)
	                            (take 20)))}
	  #+END_QUERY

	  { :query [:find (pull ?p [:block/name :block/updated-at])
	           :where
	           [?p :block/name ?name]
	           [?p :block/updated-at ?updated]
	           (not [?p :page/properties])
	           (not [?p :block/properties])
	           (not [(contains? #{"Start" "contents" "favorites" "recent"} ?name)])]
	   :result-transform (fn [result]
	                       (->> result
	                            (sort-by :block/updated-at >)
	                            (take 20)))}
